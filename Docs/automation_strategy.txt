1. OFFICIAL GERMAN GOVERNMENT RATES (ARVVwV 2025)
===============================================================================

1.1 What Are These Official Rates?
----------------------------------
Source: Auslandsreisekostenverordnung (ARV) - Foreign Travel Expenses Ordinance
Published by: Bundesministerium der Finanzen (Federal Ministry of Finance)
Updated: Annually (current: ARVVwV 2025)

Official Document: "Allgemeine Verwaltungsvorschrift über die Neufestsetzung 
der Auslandstage- und –übernachtungsgelder" (General Administrative Provision 
on Foreign Per Diem and Accommodation Allowances)

Coverage: 180+ countries and regions worldwide

1.2 What's Included in Official Rates
-------------------------------------

For Each Country, Government Specifies:

A) Daily Allowances (Tagesgeld):
   - Small per diem (>8 hours): Variable by country
   - Large per diem (24+ hours): Variable by country
   
B) Overnight Allowances (Übernachtungsgeld):
   - Maximum hotel rate per night: Variable by country
   - Different rates for major cities vs. rest of country

C) Germany Domestic Rates (2025):
   - >8 hours: 14 €
   - 24+ hours: 28 €
   - Hotel: 20 € (without receipt) or actual cost
   - Mileage: 0.30 € per km (official) or 0.88 € (you use higher rate)

4.3 Rate Lookup Function
------------------------

```javascript
async function getTravelRates(countryCode, cityName = null) {
  // First, check company custom rates
  const customRate = await db.query(`
    SELECT * FROM company_custom_rates
    WHERE country_code = $1
    AND (city_name = $2 OR city_name IS NULL)
    AND active = true
    ORDER BY city_name NULLS LAST
    LIMIT 1
  `, [countryCode, cityName]);
  
  // Then, get official government rates
  const officialRate = await db.query(`
    SELECT * FROM country_travel_rates
    WHERE country_code = $1
    AND (city_name = $2 OR city_name IS NULL)
    AND effective_from <= CURRENT_DATE
    AND (effective_until IS NULL OR effective_until >= CURRENT_DATE)
    ORDER BY city_name NULLS LAST
    LIMIT 1
  `, [countryCode, cityName]);
  
  return {
    official: officialRate.rows[0],
    custom: customRate.rows[0],
    // Use custom if available, otherwise official
    applied: {
      daily_allowance_8h: officialRate.rows[0].daily_allowance_8h,
      daily_allowance_24h: officialRate.rows[0].daily_allowance_24h,
      hotel_rate: customRate.rows[0]?.custom_hotel_rate || 
                  officialRate.rows[0].hotel_rate_max,
      mileage_rate: customRate.rows[0]?.custom_mileage_rate || 0.30,
      travel_hour_rate: customRate.rows[0]?.custom_travel_hour_rate || 98,
      work_hour_rate: customRate.rows[0]?.custom_work_hour_rate || 132
    }
  };
}
```

```javascript
async function calculateTripCosts(tripData) {
  const {
    customerId,
    engineerId,
    date,
    workHours,
    jobType
  } = tripData;
  
  // 1. Get locations
  const customer = await getCustomer(customerId);
  const engineer = await getEngineer(engineerId);
  
  // 2. Determine country and get rates
  const rates = await getTravelRates(
    customer.country_code,
    customer.city
  );
  
  // 3. Calculate distance
  const distance = await calculateDistance(
    engineer.location,
    customer.location
  );
  
  // 4. Determine if domestic or international
  const isDomestic = engineer.country === customer.country;
  const needsFlight = distance.km > 300 || !isDomestic;
  
  // 5. Get travel options
  let flightOption = null;
  let roadOption = null;
  
  if (needsFlight || distance.km > 200) {
    flightOption = await searchFlights({
      from: engineer.nearest_airport,
      to: customer.nearest_airport,
      date: date,
      returnDate: date // same day or calculate
    });
  }
  
  if (distance.km < 800 || isDomestic) {
    roadOption = await calculateRoadCosts({
      distance: distance.km,
      duration: distance.duration_minutes,
      rates: rates.applied
    });
  }
  
  // 6. Compare options and select best
  const selectedOption = compareOptions(flightOption, roadOption);
  
  // 7. Calculate duration-based costs
  const tripDuration = calculateTripDuration({
    travelTime: selectedOption.travel_hours,
    workHours: workHours,
    setupTime: 1 // hour for setup/packup
  });
  
  // 8. Calculate all cost components
  const costs = {
    // Working time
    arbeitszeit_hours: workHours,
    arbeitszeit_rate: rates.applied.work_hour_rate,
    arbeitszeit_total: workHours * rates.applied.work_hour_rate,
    
    // Travel time
    reisezeit_hours: selectedOption.travel_hours,
    reisezeit_rate: rates.applied.travel_hour_rate,
    reisezeit_total: selectedOption.travel_hours * rates.applied.travel_hour_rate,
    
    // Distance (only for road)
    entfernung_km: selectedOption.type === 'road' ? distance.km : 0,
    entfernung_rate: rates.applied.mileage_rate,
    entfernung_total: selectedOption.type === 'road' ? 
                     distance.km * rates.applied.mileage_rate : 0,
    
    // Daily allowances
    tagesspesen_24h_days: Math.floor(tripDuration.hours / 24),
    tagesspesen_24h_rate: rates.official.daily_allowance_24h,
    tagesspesen_24h_total: Math.floor(tripDuration.hours / 24) * 
                          rates.official.daily_allowance_24h,
    
    tagesspesen_8h_days: (tripDuration.hours > 8 && tripDuration.hours < 24) ? 1 : 
                         (tripDuration.hours % 24 > 8 ? 1 : 0),
    tagesspesen_8h_rate: rates.official.daily_allowance_8h,
    tagesspesen_8h_total: ((tripDuration.hours > 8 && tripDuration.hours < 24) ? 1 : 
                          (tripDuration.hours % 24 > 8 ? 1 : 0)) * 
                          rates.official.daily_allowance_8h,
    
    // Hotel
    hotel_nights: tripDuration.hours > 18 ? 
                 Math.ceil((tripDuration.hours - 8) / 24) : 0,
    hotel_rate: rates.applied.hotel_rate,
    hotel_total: (tripDuration.hours > 18 ? 
                 Math.ceil((tripDuration.hours - 8) / 24) : 0) * 
                 rates.applied.hotel_rate,
    
    // Transportation specific
    ...(selectedOption.type === 'flight' ? {
      flight_international: !isDomestic ? selectedOption.price : 0,
      flight_national: isDomestic ? selectedOption.price : 0,
      taxi: selectedOption.taxi_costs || 0,
      parken: selectedOption.parking_costs || 0
    } : {
      treibstoff: selectedOption.fuel_costs || 0,
      maut: selectedOption.toll_costs || 0
    })
  };
  
  // 9. Calculate total
  costs.komplette_rk = 
    costs.reisezeit_total +
    costs.entfernung_total +
    costs.tagesspesen_24h_total +
    costs.tagesspesen_8h_total +
    costs.hotel_total +
    (costs.flight_international || 0) +
    (costs.flight_national || 0) +
    (costs.taxi || 0) +
    (costs.parken || 0) +
    (costs.treibstoff || 0) +
    (costs.maut || 0);
  
  costs.total_quotation = costs.komplette_rk + costs.arbeitszeit_total;
  
  return {
    costs,
    selectedOption,
    rates: rates.applied,
    tripDuration
  };
}
```
